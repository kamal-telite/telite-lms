✓ Fixed docker-compose.yml (backend section)
backend:
  build:
    context: ./telite-backend
  environment:
    TELITE_DB_BACKEND: postgres
    TELITE_POSTGRES_HOST: postgres
    TELITE_POSTGRES_PORT: 5432
    TELITE_POSTGRES_DB: ${TELITE_POSTGRES_DB}          # → "telite_backend"
    TELITE_POSTGRES_USER: ${TELITE_POSTGRES_USER}      # → "telite_backend_user"
    TELITE_POSTGRES_PASSWORD: ${TELITE_POSTGRES_PASSWORD} # → separate password
The bug: Lines 55–57 in current compose use ${POSTGRES_DB}, ${POSTGRES_USER}, ${POSTGRES_PASSWORD} for the backend service. These are Moodle's bootstrap credentials. The backend ends up connecting to the moodle database with moodleuser.
The fix: Change to ${TELITE_POSTGRES_DB}, ${TELITE_POSTGRES_USER}, ${TELITE_POSTGRES_PASSWORD}. These resolve to the backend's own credentials, ensuring complete DB isolation.
📄 docker/init/01-create-databases.sql (NEW)
-- This script runs on first Postgres container initialization only.
-- Mount into /docker-entrypoint-initdb.d/ in docker-compose.yml

-- 1. Moodle database & user
CREATE DATABASE moodle;
CREATE USER moodleuser WITH PASSWORD 'moodlepass';
GRANT ALL PRIVILEGES ON DATABASE moodle TO moodleuser;

-- 2. Backend database & user
CREATE DATABASE telite_backend;
CREATE USER telite_backend_user WITH PASSWORD 'change_me_backend_pass';
GRANT ALL PRIVILEGES ON DATABASE telite_backend TO telite_backend_user;

-- 3. Revoke cross-access
REVOKE ALL ON DATABASE moodle FROM telite_backend_user;
REVOKE ALL ON DATABASE telite_backend FROM moodleuser;
